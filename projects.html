Playwright — JavaScript (POM)

Exemplo mínimo de estrutura e teste.

// samples/playwright/example.spec.js
const { test, expect } = require('@playwright/test');
const LoginPage = require('./pages/login.page');

test('Deve logar com credenciais válidas', async ({ page }) => {
  const login = new LoginPage(page);
  await login.goto();
  await login.login('user@example.com','senha123');
  await expect(page).toHaveURL(/dashboard/);
});

Cypress — JavaScript

Exemplo de teste Cypress simples.

// samples/cypress/specs/login.cy.js
describe('Login flow', () => {
  it('logs in successfully', () => {
    cy.visit('https://example.com/login');
    cy.get('[data-test=email]').type('user@example.com');
    cy.get('[data-test=password]').type('senha123');
    cy.get('[data-test=submit]').click();
    cy.url().should('include', '/dashboard');
  });
});

API Testing — JavaScript & Java

Exemplo com supertest (Node.js) e esboço Rest-Assured (Java).

// samples/api/supertest.test.js
const request = require('supertest');
const api = request('https://api.example.com');

test('GET /status returns 200', async () => {
  const res = await api.get('/status').set('Accept','application/json');
  expect(res.status).toBe(200);
  expect(res.body).toHaveProperty('status','ok');
});
